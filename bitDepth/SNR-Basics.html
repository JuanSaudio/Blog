Signal to Noise Ratio Basic Explanation¶

# Section to import different libraries
import numpy as np
import matplotlib.pyplot as plt

N = 10000
n = np.arange(N)
f = 5
x = np.sin(2 * np.pi * f * n / N)
y = np.random.normal(size=N) * 0.01

plt.plot(n, y)
plt.plot(n, x)
plt.grid(True)
plt.autoscale(tight=True)
